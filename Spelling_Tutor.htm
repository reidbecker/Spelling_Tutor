<script>

// --- CONFIGURATION ---

// !!! CRITICAL: Replace the placeholder below with the actual URL from your Make.com Webhook !!!

const MAKE_WEBHOOK_URL = 'https://hook.us2.make.com/khlm9x843q59bwonukw1yxbls4zu9pst';

// --- STATE ---

let chatHistory = [];

let recognition; // For Web Speech API - Voice to Text

let isListening = false;

let isUserIdEntered = false;

 

// --- DOM Elements ---

const chatWindow = document.getElementById('chat-window');

const userInput = document.getElementById('user-input');

const sendButton = document.getElementById('send-button');

const micButton = document.getElementById('mic-button');

const idInput = document.getElementById('user-id');

const statusMessage = document.getElementById('status-message');

const warningMessage = document.getElementById('warning-message');

 

// --- UTILITY FUNCTIONS ---

 

function displayMessage(text, isTutor) {

const rowClass = isTutor ? 'tutor-row' : 'student-row';

const messageClass = isTutor ? 'tutor-message' : 'student-message';

const messageRow = document.createElement('div');

messageRow.className = 'message-row ' + rowClass;

 

const messageDiv = document.createElement('div');

messageDiv.className = 'message ' + messageClass;

messageDiv.textContent = text;

messageRow.appendChild(messageDiv);

chatWindow.appendChild(messageRow);

// Auto-scroll to the bottom

chatWindow.scrollTop = chatWindow.scrollHeight;

}

 

function speakText(text) {

// Check if Web Speech Synthesis is supported

if ('speechSynthesis' in window) {

const utterance = new SpeechSynthesisUtterance(text);

// Optionally set a friendly, clear voice (will depend on user's OS/browser)

utterance.voice = speechSynthesis.getVoices().find(v => v.lang === 'en-US' && v.name.includes('Google') || v.name.includes('Zira'));

// Clear the queue and speak

speechSynthesis.cancel();

speechSynthesis.speak(utterance);

}

}

function updateControls(isDisabled) {

userInput.disabled = isDisabled;

sendButton.disabled = isDisabled;

micButton.disabled = isDisabled || !('webkitSpeechRecognition' in window);

}

 

// --- CORE LOGIC ---

 

async function sendMessage(input, isVoice = false) {

if (!isUserIdEntered) {

displayMessage("Please enter your Student ID before starting.", true);

return;

}

if (!input.trim()) return;

 

// 1. Display and speak student input

displayMessage(input, false);

updateControls(true); // Disable input while waiting for response

userInput.value = '';

 

// 2. Prepare and send payload to Make.com

const userId = idInput.value;

const payload = {

user_id: userId,

user_input: input,

chat_history: JSON.stringify(chatHistory)

};

 

// 3. Update chat history with student's message for next turn

chatHistory.push({ role: "user", text: input });

 

try {

// Check if the placeholder URL is still present (or corrupted)

if (MAKE_WEBHOOK_URL.includes('[YOUR_MAKE_WEBHOOK_URL_HERE]')) {

statusMessage.style.display = 'block';

statusMessage.textContent = "!!! CRITICAL ERROR: Please replace the placeholder MAKE_WEBHOOK_URL in the JavaScript code with your actual MAKE.com Webhook URL.";

updateControls(true);

return;

}

const response = await fetch(MAKE_WEBHOOK_URL, {

method: 'POST',

headers: { 'Content-Type': 'application/json' },

body: JSON.stringify(payload)

});

 

if (!response.ok) {

throw new Error(`HTTP error! status: ${response.status}`);

}

 

const data = await response.json();

const tutorResponse = data.tutor_response;

// 4. Handle Tutor Response

if (tutorResponse) {

displayMessage(tutorResponse, true);

speakText(tutorResponse);

// 5. Update chat history with tutor's message

chatHistory.push({ role: "tutor", text: tutorResponse });

}

 

} catch (error) {

console.error('Error communicating with Make/ElevenLabs:', error);

const errorMessage = "Sorry, I had trouble connecting to the network. Please try again.";

displayMessage(errorMessage, true);

speakText(errorMessage);

} finally {

updateControls(false); // Enable input again

}

}

 

// --- VOICE RECOGNITION (MIC BUTTON) SETUP ---

 

function setupVoiceRecognition() {

if ('webkitSpeechRecognition' in window) {

recognition = new webkitSpeechRecognition();

recognition.continuous = false; // Only recognize one phrase

recognition.interimResults = false; // Only return final result

recognition.lang = 'en-US';

 

recognition.onstart = () => {

isListening = true;

micButton.textContent = '🛑 Stop Listening';

micButton.style.backgroundColor = '#d32f2f'; // Change color while listening

};

 

recognition.onend = () => {

isListening = false;

micButton.textContent = '🎤 Speak';

micButton.style.backgroundColor = '#ff5722';

};

 

recognition.onresult = (event) => {

const transcript = event.results[0][0].transcript;

if (transcript) {

sendMessage(transcript, true);

}

};

recognition.onerror = (event) => {

console.error('Speech Recognition Error:', event.error);

if (event.error !== 'no-speech') {

displayMessage("Microphone error: Please check your settings and try again.", true);

}

};

} else {

// Browser does not support Speech Recognition

micButton.disabled = true;

micButton.style.backgroundColor = '#cccccc';

warningMessage.style.display = 'block';

warningMessage.textContent = "Your browser does not fully support voice input. Please use the typing field (Google Chrome recommended for full voice features).";

}

}

 

// --- EVENT HANDLERS ---

// Handle User ID Input and Control State

idInput.addEventListener('input', () => {

const id = idInput.value.trim();

const isValid = id.length >= 1; // Basic validation: ID must be at least 1 character

 

if (isValid && !isUserIdEntered) {

// First time a valid ID is entered, enable controls

isUserIdEntered = true;

updateControls(false);

idInput.disabled = true; // Lock the ID once conversation starts

// Initiate the first message to the Tutor

displayMessage("Welcome! Starting new session...", true);

sendMessage("Hello Tutor, please introduce yourself and ask me for my grade level.", true);

} else if (!isValid) {

// If ID is cleared or invalid before starting

isUserIdEntered = false;

updateControls(true);

}

});

 

// Handle Send Button Click

sendButton.addEventListener('click', () => {

sendMessage(userInput.value);

});

 

// Handle Enter Key Press in input field

userInput.addEventListener('keypress', (e) => {

if (e.key === 'Enter') {

sendMessage(userInput.value);

}

});

 

// Handle Mic Button Click

micButton.addEventListener('click', () => {

if (isListening) {

recognition.stop();

} else {

// Stop any speaking and start listening

speechSynthesis.cancel();

recognition.start();

}

});

 

// --- INITIALIZATION ---

window.onload = () => {

setupVoiceRecognition();

// Check for the CRITICAL ERROR placeholder on load

if (MAKE_WEBHOOK_URL.includes('[YOUR_MAKE_WEBHOOK_URL_HERE]')) {

statusMessage.style.display = 'block';

statusMessage.textContent = "!!! CRITICAL ERROR: Please replace the placeholder MAKE_WEBHOOK_URL in the JavaScript code with your actual MAKE.com Webhook URL.";

updateControls(true);

} else {

// If URL is set, hide error and enable ID input

document.getElementById('user-id').disabled = false;

idInput.focus();

}

};
